let test = { "a.b.c": 1, "a.d": 2, e: 3 };
let getObj = function (obj) {
	let newObj = {};
	let helper = function (parts, val, obj) {
		if (parts.length === 0) return val;
		let currKey = parts.shift();
		if (!obj.hasOwnProperty(currKey)) {
			obj[currKey] = {};
		}
		obj[currKey] = helper(parts, val, obj[currKey]);
		return obj;
	};
	for (let key in obj) {
		const val = obj[key];
		const parts = key.split(".");
		helper(parts, val, newObj);
	}
	console.log(newObj);
};
getObj(test);
