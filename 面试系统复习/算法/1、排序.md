# 快排
- 空间：n
- 时间：n logn
- 无序
```js
function sortArray(nums) {
	function quickSort(start, end) {
		if (start >= end) return;
		const pivot = start + Math.floor(Math.random() * (end - start + 1));
		swap(pivot, start);
		let i = start + 1;
		let j = end;
		while (i < j) {
			while (nums[start] > nums[i]) {
				i++;
			}
			while (nums[start] < nums[j]) {
				j--;
			}
			if (i < j) {
				swap(i, j);
				i++;
				j--;
			} else {
				break;
			}
		}
		swap(start, j);
		quickSort(start, j - 1);
		quickSort(i, end);
	}
	function swap(i, j) {
		[nums[i], nums[j]] = [nums[j], nums[i]];
	}
	quickSort(0, nums.length - 1);
	return nums;
}
```
